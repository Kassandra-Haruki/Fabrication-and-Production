mission "ref_Refinement"
	non-blocking
	landing
	invisible
    repeat
	to offer
        has "outfit (installed): ref_refinery"
        and
            "outfit: Aluminum" > 1
            has "outfit: ref_blueprint_Aluminum"
    on offer
        conversation
            action
                "ref_Aluminum" = "outfit: Aluminum" / 2
                "ref_remainderAluminum" = "outfit: Aluminum" % 2
                "ref_AluminumOre" = "outfit: Aluminum" - "ref_remainderAluminum"
                "ref_valueAluminum" = "refAluminumOre" * 1800
                "ref_valueAluminumBar" = "ref_valueRawAluminum" * 3 #because the value of 1 bar = 3 ores
            `Your Refinery turned:`
                `&[ref_AluminumOre] Aluminum Ores (worth &[credits@ref_valueAluminum] credits) into &[ref_Aluminum] Aluminum Bars (worth &[credits@ref_valueAluminumBar] credits)!`
                    [to display]
                        "has: ref_Aluminum"
            label "refine Aluminum"
            branch "refine Copper"
                not "ref_blueprint_Aluminum"
                "outfit: Aluminum" < 2
            action
                outfit "Aluminum" -2
                outfit "ref_Aluminum Bar"
            branch "refine Aluminum"
            label "refine Copper"
                decline
